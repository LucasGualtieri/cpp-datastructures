576 void export_to_png(const std::string& filename, const std::string& engine = "dot", size_t dpi = 300) const {
577
578 // NOTE: This is possible to define all nodes and edges at once
579 // node [shape=ellipse, style=filled, fillcolor=lightblue];
580 // edge [color=gray, style=dotted];
581
582 // 1. Create a DOT file representing the graph
583 std::string dotFilename = filename + ".dot";
584 std::ofstream dotFile(dotFilename);
585
586 if (!dotFile) {
587 throw std::runtime_error("Could not create the DOT file.");
588 }
589
590 dotFile << (directed ? "digraph" : "graph") << " G {\n";
591
592 dotFile << std::format(" dpi = {};", dpi) << std::endl;
593
594 // for (const auto& [v, propsObject] : vertexMap) {
595 for (const Vertex& v : vertices()) {
596 // auto [color, line] = propsObject;
597 auto [color, line] = vertexMap.Find(v);
598 std::string label = vertexLabel.Find(v) == "" ? std::format("{}", v) : vertexLabel.Find(v);
599 std::string props = "color = \"" + color + "\", style = \"" + line;
600 dotFile << " " << v << " [ " << props << format("\", label=\"{}\"]", label) << ";\n";
601 }
602
603 dotFile << std::endl;
604
605 for (const Edge& e : edges()) {
606
607 if (!directed && e.u > e.v) continue; // Avoid duplicate edges for undirected graphs
608
609 Prop prop = edgeMap.Find({e.u, e.v});
610 auto [color, line] = prop == base && !directed ? edgeMap.Find({e.v, e.u}) : prop;
611
612 std::string props = "color = \"" + color + "\", style = \"" + line;
613
614 dotFile << " " << e.u << (directed ? " -> " : " -- ") << e.v << " [label=\"" << (weighted ? std::format("{:.6g}", e.weight) : "") << "\", " << props << "\"];\n";
615 }
616
617 dotFile << "}\n";
618 dotFile.close();
619
620 // 2. Use the system() function to run the Graphviz command
621 std::string command = engine + " -Tpng " + dotFilename + " -o " + filename + ".png";
622
623 if (system(command.c_str()) != 0) {
624 throw std::runtime_error("Failed to execute Graphviz command.");
625 }
626
627 std::cout << "Graph exported successfully to " << filename << ".png\n";
628 }

672 double density(const int& precision = 3) const {
673
674 double n = this->n;
675 double m = this->m;
676
677 double density = m / ((n * n - n) / (directed ? 1 : 2));
678
679 double factor = std::pow(10, precision);
680 return std::round(density * factor) / factor;
